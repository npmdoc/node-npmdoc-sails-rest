function sails-rest.registerConnection (connection, collections, cb)

  • description and source-code
    registerConnection = function (connection, collections, cb) {
  var config = {
    hooks: {},
    methods: {
      create: 'post',
      find: 'get',
      update: 'put',
      destroy: 'del'
    },
    headers: connection.headers || {}
  };
  if(!connection.identity) return cb(new Error('Connection is missing an identity.'));
  if(connections[connection.identity]) return cb(new Error('Connection is already registered.'));

  // Add in logic here to initialize connection
  // e.g. connections[connection.identity] = new Database(connection, collections);
  config.endpoint = url.format({
    host: connection.host,
    pathname: connection.pathname,
    protocol: connection.protocol
  });
  config.http = connection.http || {};
  config.methods = _.extend(config.methods, (connection.methods || {}));

  if(connection.hooks.merge) {
    config.hooks.before = hooks.before.concat(connection.hooks.before || []);
    config.hooks.after = hooks.after.concat(connection.hooks.after || []);
  } else {
    config.hooks.before = connection.hooks.before || [];
    config.hooks.after = connection.hooks.after || [];
  }

  connections[connection.identity] = config;

  cb();
}

function sails-rest.teardown (conn, cb)

  • description and source-code
    teardown = function (conn, cb) {

  if (typeof conn == 'function') {
    cb = conn;
    conn = null;
  }
  if (!conn) {
    connections = {};
    return cb();
  }
  if(!connections[conn]) return cb();
  delete connections[conn];
  cb();
}

function sails-rest.connection.create (connection, collection, values, cb)

  • description and source-code
    create = function (connection, collection, values, cb) {
  var config = {},
    req,
    conn = {
      connection: connection,
      collection: collection,
      options: {},
      values: values
    };
  runBeforeHooks(req, 'create', config, conn, function (err, req) {
    if(err) {
      return cb(err);
    }
    return req.send(conn.values).end(getResponseHandler(conn.connection, cb));
  });

function sails-rest.connection.find (connection, collection, options, cb)

  • description and source-code
    find = function (connection, collection, options, cb) {
  var config = {},
      req,
      conn = {
        connection: connection,
        collection: collection,
        options: options,
        values: {}
      };
  runBeforeHooks(req, 'find', config, conn, function (err, req) {
    if(err) {
      return cb(err);
    }
    return req.query(conn.options).end(function (err, res) {
      if (res) {
        err = null;
      }
      // Waterline requires that `find` will return an array.
      // Here we ensure that response body is converted to an array if required.
      if(!err){
        if(!Array.isArray(res.body)){
          res.body = [res.body];
        }
      }
      handleResponse(conn.connection, err, res, cb);
    });
  });
}
